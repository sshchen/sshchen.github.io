    5. Binary Search 
    class Solution {
    public int search(int[] nums, int target) {
        int i = 0, j = nums.length - 1;
        while (i <= j) {
            int m = i + (j - i) / 2; // (j + i) / 2 migth cause integer out of bound
            if (nums[m] == target) {
                return m;
            }
            if (nums[m] < target) {
                i = m + 1;
            } else {
                j = m - 1;
            }
        }
        return -1;
    }
}
    • Time: O(logn)
    • Space: O(1)
